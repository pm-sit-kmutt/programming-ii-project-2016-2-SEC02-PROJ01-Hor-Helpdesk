int dormId = 0;
        try {
            Connection connect = ConnectionBuilder.getConnection();
            PreparedStatement ps = connect.prepareStatement(
                "SELECT User.userId, Dormitory.dormId " +
"                FROM User " +
"                JOIN Dormitory ON User.userId = Dormitory.User_userId " +
"                WHERE userId = ? ");
            ps.setLong(1, userId);
            ResultSet result = ps.executeQuery();
            while(result.next()){
                dormId = result.getInt("dormId");
            }
            ps.close();
            ps = connect.prepareStatement(
                "SELECT Problem_problemId, SUM(RecordProblem.problemCurrent), SUM(RecordProblem.problemPast) "
                + "FROM Cause "
                + "JOIN RecordProblem ON RecordProblem.Cause_causeId = Cause.causeId "
                + "JOIN Room ON Room.roomId = Cause.Room_roomId "
                + "WHERE Room.Dormitory_dormId = ? "
                + "GROUP BY Problem_problemId");
            ps.setInt(1, dormId);
            result = ps.executeQuery();
            while(result.next()){
                current[result.getInt("Problem_problemId")-1] = result.getInt("SUM(RecordProblem.problemCurrent)");
                past[result.getInt("Problem_problemId")-1] = result.getInt("SUM(RecordProblem.problemPast)");
            }
            ps.close();
            connect.close();
        }
        catch(SQLException | ClassNotFoundException e) {
            e.printStackTrace();
        }
        catch(Exception e) {
            e.printStackTrace();
        }
